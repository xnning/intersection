\colorlet{wfcolor}{BrickRed}


\newcommand{\ruleLabelSourceWF}{\bm{\textcolor{wfcolor}{wf}}}

\newcommand{\ruleLabelSourceWFVar}{\ruleLabelSourceWF\text{var}}
\newcommand{\ruleLabelSourceWFTop}{\ruleLabelSourceWF\text{top}}
\newcommand{\ruleLabelSourceWFFun}{\ruleLabelSourceWF\text{fun}}
\newcommand{\ruleLabelSourceWFForall}{\ruleLabelSourceWF\text{forall}}
\newcommand{\ruleLabelSourceWFAnd}{\ruleLabelSourceWF{\text{and}}}
\newcommand{\ruleLabelSourceWFRec}{\ruleLabelSourceWF\text{rec}}

\newcommand{\ruleSourceWF}{
\inferrule* [right=$\ruleLabelSourceWF$]
  {\ftv \A \in \gamma}
  {\istype \gamma \A}
}

\newcommand{\ruleLabelTargetWF}{\ruleLabelTarget\text{wf}}
\newcommand{\ruleTargetWF}{
\inferrule* [right=$\ruleLabelTargetWF$]
  {\ftv T \in \Gamma}
  {\judgeTargetWF \Gamma T}
}

% Expanded form of well-formedness

\newcommand{\ruleSourceWFVar}{
\inferrule* [right=$\ruleLabelSourceWFVar$]
  {\alpha \in \gamma}
  {\istype \gamma \alpha}
}

\newcommand{\ruleSourceWFTop}{
\inferrule* [right=$\ruleLabelSourceWFTop$]
  { }
  {\istype \gamma \top}
}

\newcommand{\ruleSourceWFFun}{
\inferrule* [right=$\ruleLabelSourceWFFun$]
  {\istype \gamma {\A_1} \\ \istype \Gamma {\A_2}}
  {\istype \gamma {\A_1 \to \A_2}}
}

\newcommand{\ruleSourceWFForall}{
\inferrule* [right=$\ruleLabelSourceWFForall$]
  {\istype {\gamma, \alpha} \A}
  {\istype \gamma {\for \alpha \A}}
}

\newcommand{\ruleSourceWFAnd}{
\inferrule* [right=$\ruleLabelSourceWFAnd$]
  {\istype \gamma {\A_1} \\ \istype \Gamma {\A_2}}
  {\istype \gamma {\A_1 \intersect \A_2}}
}

\newcommand{\ruleSourceWFRec}{
\inferrule* [right=$\ruleLabelSourceWFRec$]
  {\istype \gamma \A}
  {\istype \gamma {\recordType l \A}}
}
