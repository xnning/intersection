\newcommand{\formdis}{\framebox{$ \jdis \Gamma A B $}}

\newcommand{\makelabeldis}[1]{D$#1$}

\newcommand{\labeldistop}{\makelabeldis \top}
\newcommand{\ruledistop}{
  \inferrule* [right=\labeldistop]
    { }
    {\jdis \Gamma \top A}
}

\newcommand{\labeldistopsym}{\makelabeldis {\top Sym}}
\newcommand{\ruledistopsym}{
  \inferrule* [right=\labeldistopsym]
    { }
    {\jdis \Gamma A \top}
}

\newcommand{\labeldisvar}{\makelabeldis \alpha}
\newcommand{\ruledisvar}{
  \inferrule* [right=\labeldisvar]
    {\alpha \disjoint A \in \Gamma \\ A \subtype B}
    {\jdis \Gamma \alpha B}
}

\newcommand{\labeldissym}{\makelabeldis {\alpha Sym}}
\newcommand{\ruledissym}{
  \inferrule* [right=\labeldissym]
    {\alpha \disjoint A \in \Gamma \\ A \subtype B}
    {\jdis \Gamma B \alpha}
}

\newcommand{\labeldisinterl}{\makelabeldis {\inter L}}
\newcommand{\ruledisinterl}{
  \inferrule* [right=\labeldisinterl]
    {\jdis \Gamma {A_1} B \\ \jdis \Gamma {A_2} B}
    {\jdis \Gamma {A_1 \inter A_2} {B}}
}

\newcommand{\labeldisinterr}{\makelabeldis {\inter R}}
\newcommand{\ruledisinterr}{
  \inferrule* [right=\labeldisinterr]
    {\jdis \Gamma A {B_1} \\ \jdis \Gamma A {B_2}}
    {\jdis \Gamma {A} {B_1 \inter B_2}}
}

\newcommand{\labeldisfun}{\makelabeldis \rightarrow}
\newcommand{\ruledisfun}{
  \inferrule* [right=\labeldisfun]
    {\jdis \Gamma {A_2} {B_2}}
    {\jdis \Gamma {A_1 \to A_2} {B_1 \to B_2}}
}

\newcommand{\labeldisforall}{\makelabeldis \forall}
\newcommand{\ruledisforall}{
  \inferrule* [right=\labeldisforall]
    {\jdis {\Gamma, \alpha } {B} {C}}
    {\jdis \Gamma {\for \alpha {B}} {\for \alpha {C}}}
}

\newcommand{\ruledisforalldis}{
  \inferrule* [right=\labeldisforall]
    {\jdis {\Gamma, \alpha * A} {B} {C}}
    {\jdis \Gamma {\fordis \alpha {A} {B}} {\fordis \alpha {A} {C}}}
}

\newcommand{\ruledisforallext}{
  \inferrule* [right=\labeldisforall]
    {\jdis {\Gamma, \alpha \disjoint {{A_1} \inter {A_2}}} {B} {C}}
    {\jdis \Gamma {\fordis \alpha {A_1} {B}} {\fordis \alpha {A_2} {C}}}
}

\newcommand{\labeldisreceq}{\makelabeldis {Rec_=}}
\newcommand{\ruledisreceq}{
  \inferrule* [right=\labeldisreceq]
    {\jdis \Gamma A B}
    {\jdis \Gamma {\recordType l A} {\recordType l B}}
}

\newcommand{\labeldisrecneq}{\makelabeldis {Rec_{\neq}}}
\newcommand{\ruledisrecneq}{
  \inferrule* [right=\labeldisrecneq]
    {l_1 \neq l_2}
    {\jdis \Gamma {\recordType {l_1} A} {\recordType {l_2} B}}
}

\newcommand{\labeldisax}{\makelabeldis Ax}
\newcommand{\ruledisatomic}{
  \inferrule* [right=\labeldisax]
    {A \disjointax B}
    {\jdis \Gamma A B}
}


\newcommand{\formax}{\framebox{$ A \disjointax B$}}

\newcommand{\makelabelax}[1]{$\text{D}_{Ax}#1$}

\newcommand{\labelaxintforall}{\makelabelax {(\mathbb{Z} \forall)}}
\newcommand{\ruleaxintforall}{
  \inferrule* [right=\labelaxintforall]
    { }
    {\tyint \disjointax \for \alpha B}
}

\newcommand{\ruleaxintforalldis}{
  \inferrule* [right=\labelaxintforall]
    { }
    {\tyint \disjointax \fordis \alpha {B_1} {B_2}}
}

\newcommand{\labelaxintfun}{\makelabelax {(\mathbb{Z} \! \rightarrow)}}
\newcommand{\ruleaxintfun}{
  \inferrule* [right=\labelaxintfun]
    { }
    {\tyint \disjointax {A_1} \to {A_2}}
}

\newcommand{\labelaxintrec}{\makelabelax {(\mathbb{Z} Rec)}}
\newcommand{\ruleaxintrec}{
  \inferrule* [right=\labelaxintrec]
    { }
    {\tyint \disjointax {\recordType l A}}
}

\newcommand{\labelaxfunforall}{\makelabelax {(\rightarrow \! \forall)}}
\newcommand{\ruleaxfunforall}{
  \inferrule* [right=\labelaxfunforall]
    { }
    {A_1 \to A_2 \disjointax \for \alpha B}
}

\newcommand{\ruleaxfunforalldis}{
  \inferrule* [right=\labelaxfunforall]
    { }
    {A_1 \to A_2 \disjointax \fordis \alpha {B_1} {B_2}}
}

\newcommand{\labelaxfunrec}{\makelabelax {(\rightarrow \! Rec)}}
\newcommand{\ruleaxfunrec}{
  \inferrule* [right=\labelaxfunrec]
    { }
    {A_1 \to A_2 \disjointax {\recordType l B}}
}

\newcommand{\labelaxforalldisrec}{\makelabelax {(\forall Rec)}}
\newcommand{\ruleaxforalldisrec}{
  \inferrule* [right=\labelaxforalldisrec]
    { }
    {\fordis \alpha {A_1} {A_2} \disjointax {\recordType l B}}
}

\newcommand{\labelaxsym}{\makelabelax Sym}
\newcommand{\ruleaxsym}{
  \inferrule* [right=\labelaxsym]
    {B \disjointax A}
    {A \disjointax B}
}
