\begin{figure}
    \begin{mathpar}
        \inferrule* [right=$\rulelabel{bool}$]
            { }
            {\typing {\boldcdot} {\true, \false} {\bool} {bool}}

        \inferrule* [right=$\rulelabel{int}$]
            { }
            {\typing {\boldcdot} {\ldots, -2, -1, 0, 1, \ldots} {\integer} {int}}

        \inferrule* [right=$\rulelabel{var}$]
            {x : \tau \in \Gamma}
            {\typing \Gamma x \tau {var}}

        \inferrule* [right=$\rulelabel{lam}$]
            {\typing {\Gamma, x : \tau} {e} {\tau_1} {l}}
            {\typing \Gamma {\lam {x : \tau} {e}} {\tau \to \tau_1} {lam(l)}}

        \inferrule* [right=$\rulelabel{app}$]
            {\typing \Gamma {e_1} {\tau_2 \to \tau_1} {l_1} \\
             \typing \Gamma {e_2} {\tau_2} {l_1}}
            {\typing \Gamma {\app {e_1} {e_2}} {\tau_1} {app(l_1, l_2)}}

        \inferrule* [right=$\rulelabel{merge}$]
            {\typing \Gamma {e_1} {\tau_1} {l_1} \\
             \typing \Gamma {e_2} {\tau_2} {l_2}}
            {\typing \Gamma {e_1 \mergeop e_2} {\tau_1 \intersectionop \tau_2} {merge(l_1, l_2)}}
    \end{mathpar}
    \caption{Type system.}
\end{figure}
