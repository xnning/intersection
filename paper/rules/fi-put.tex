\framebox{\( \turnsput (t; l; e : t \yieldsnothing E) : (t, t) \yieldsnothing C \)}

\infax[PutBase]
{\turnsput (\recordtype l t; l; e : t' \yieldsnothing E) : (t, \recordtype l {t'})
  \yieldsnothing {\abs {\rel x {\image {\recordtype l t}}} E}}

\infrule[PutLeft]
{\turnsput (t_1; l; e : t \yieldsnothing E) : (t', t_1') \yieldsnothing C}
{\turnsput (t_1 \with t_2; l; e : t \yieldsnothing E) : (t', t_1' \with t_2)
  \yieldsnothing {\abs {\rel x {\image {t_1 \with t_2}}} {C (\fst ~ x)}}}

\infrule[PutRight]
{\turnsput (t_2; l; e : t \yieldsnothing E) : (t', t_2') \yieldsnothing C}
{\turnsput (t_1 \with t_2; l; e : t \yieldsnothing E) : (t', t_1' \with t_2)
  \yieldsnothing {\abs {\rel x {\image {t_1 \with t_2}}} {C (\fst ~ x)}}}
