% Type well-formedness in the System F_&

\newcommand{\ruleewfvar}{
\inferrule* 
  {\alpha \in \gamma}
  {\judgeewf \gamma \alpha}
}

\newcommand{\ruleewffun}{
\inferrule* 
  {\judgeewf \gamma \tau_1 \andalso \judgeewf \gamma \tau_2}
  {\judgeewf \gamma {\tau_1 \to \tau_2}}
}

\newcommand{\ruleewfforall}{
\inferrule* 
  {\judgeewf {\gamma, \alpha} \tau}
  {\judgeewf \gamma {\for \alpha \tau}}
}

\newcommand{\ruleewfand}{
\inferrule* 
  {\judgeewf \gamma \tau_1 \andalso \judgeewf \gamma \tau_2}
  {\judgeewf \gamma {\tau_1 \andop \tau_2}}
}

\newcommand{\ruleewfrec}{
\inferrule* 
  {\judgeewf \gamma \tau}
  {\judgeewf \gamma {\recty l \tau}}
}

% Type well-formedness in the target system

\newcommand{\ruletwfvar}{
\inferrule* 
  {\alpha \in T}
  {\judgetwf T \alpha}
}

\newcommand{\ruletwffun}{
\inferrule* 
  {\judgetwf \Gamma T_1 \andalso \judgetwf \Gamma T_2}
  {\judgetwf \Gamma {T_1 \to T_2}}
}

\newcommand{\ruletwfforall}{
\inferrule* 
  {\judgetwf {\Gamma, \alpha} T}
  {\judgetwf \Gamma {\for \alpha T}}
}

\newcommand{\ruletwfpair}{
\inferrule* 
  {\judgetwf \Gamma T_1 \andalso \judgetwf \Gamma T_2}
  {\judgetwf \Gamma {\pair {T_1} {T_2}}}
}
