\framebox{\( \turnsget (t; l) : t \yieldsnothing C \)}

\infax[GetBase]
{\turnsget (\recordtype l t; l) : t
  \yieldsnothing {\abs {\rel x {\image {\recordtype l t}}} x}}

\infrule[GetLeft]
{\turnsget (t_1; l) : t \yieldsnothing C}
{\turnsget (t_1 \with t_2; l) : t
  \yieldsnothing {\abs {\rel x {\image {t_1 \with t_2}}} {C (\app \fst x)}}}

\infrule[GetRight]
{\turnsget (t_2; l) : t \yieldsnothing C}
{\turnsget (t_1 \with t_2; l) : t
  \yieldsnothing {\abs {\rel x {\image {t_1 \with t_2}}} {C (\app \snd x)}}}
