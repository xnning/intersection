\infax[GetBase]
{\turnsget (\recordtype l \ty; l) : \ty
  \yieldsnothing {\idmono {\image {\recordtype l \ty}}}}

{\turnsget (\ty_1; l) : \ty \yieldsnothing C}
{\turnsget (\ty_1 \intersects \ty_2; l) : \ty
  \yieldsnothing {\absty x {\image {\ty_1 \intersects \ty_2}} {C (\app \fst x)}}}

\infrule[GetRight]
{\turnsget (\ty_2; l) : \ty \yieldsnothing C}

  \yieldsnothing {\absty x {\image {\ty_1 \intersects \ty_2}} {C (\app \snd x)}}}
