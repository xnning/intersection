\framebox{\( \ty \subtype \ty \yields C \)}

\inferrule [SVar]
{}
{\alpha \subtype \alpha \yields {\Lam x {\image \alpha} x}}

\inferrule [SFun]
{\ty_3 \subtype \ty_1 \yields {C_1} \andalso \ty_2 \subtype \ty_4 \yields {C_2}}
{\ty_1 \to \ty_2 \subtype \ty_3 \to \ty_4
  \yields
    {\Lam f {\image {\ty_1 \to \ty_2}}
      {\Lam x {\image {\ty_3}}
        {\App {C_2} {(\App f {(\App {C_1} x)})}}}}}

\inferrule [SForall]
{\ty_1 \subtype \subst {\alpha_2} {\alpha_1} \ty_2 \yields C}
{\Forall \alpha_1 \ty_1 \subtype \Forall \alpha_2 \ty_2
  \yields
    {\Lam f {\image {\Forall \alpha \ty_1}}
      {\BLam \alpha {\App C {(\App f \alpha)}}}}}

\inferrule [$ \Intersect_k $]
{\ty_k \subtype \ty_3 \yields C}
{\ty_1 \Intersect \ty_2 \subtype \ty_3
  \yields
    {\Lam x {\image {\ty_1 \Intersect \ty_2}}
      {\App C {(\Proj k x)}}}}

\inferrule [SAnd3]
{\ty_1 \subtype \ty_2 \yields {C_1} \andalso \ty_1 \subtype \ty_3 \yields {C_2}}
{\ty_1 \subtype \ty_2 \Intersect \ty_3
  \yields
    {\Lam x {\image {\ty_1}}
      {\Pair {\App {C_1} x} {\App {C_2} x}}}}

\inferrule [SRec]
{\ty_1 \subtype \ty_2 \yields C}
{\RecTy l {\ty_1} \subtype \RecTy l {\ty_2}
  \yields
    {\Lam x {\image {\RecTy l {\ty_1}}} {\App C x}}}
