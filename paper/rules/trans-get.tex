% The field with label $ l $ inside $ \ty_1 $ is of type $ \ty_2 $.

  {}
  {\turnsget (\RecTy l \ty; l) : \ty \yields {\Lam x {\image {\RecTy l \ty}} x}}

{\turnsget (\ty_1; l) : \ty \yields C}
{\turnsget (\ty_1 \Intersect \ty_2; l) : \ty
  \yields {\Lam x {\image {\ty_1 \Intersect \ty_2}} {C (\Proj k x)}}}
