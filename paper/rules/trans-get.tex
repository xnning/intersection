\infax[GetBase]
{\turnsget (\recordtype l \ty; l) : \ty
  \yields {\idmono {\image {\recordtype l \ty}}}}

\infrule[GetLeft]
{\turnsget (\ty_1; l) : \ty \yields C}
{\turnsget (\ty_1 \with \ty_2; l) : \ty
  \yields {\absty x {\image {\ty_1 \with \ty_2}} {C (\app \fst x)}}}

\infrule[GetRight]
{\turnsget (\ty_2; l) : \ty \yields C}
{\turnsget (\ty_1 \with \ty_2; l) : \ty
  \yields {\absty x {\image {\ty_1 \with \ty_2}} {C (\app \snd x)}}}
