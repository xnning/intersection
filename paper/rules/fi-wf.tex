\framebox{\(\Gamma \vdash {t}\)}

\infrule[WF-Var]
{\alpha \in \Gamma}
{\Gamma \vdash \alpha}

\infrule[WF-Fun]
{\Gamma \vdash t_1 \andalso \Gamma \vdash t_2}
{\Gamma \vdash t_1 \to t_2}

\infrule[WF-Forall]
{\Gamma, \alpha \vdash t}
{\Gamma \vdash \forall \alpha. t}

\infrule[WF-And]
{\Gamma \vdash t_1 \andalso \Gamma \vdash t_2}
{\Gamma \vdash t_1 \with t_2}

\infrule[WF-Rcd]
{\Gamma \vdash t}
{\Gamma \vdash \recordtype{l}{t}}
