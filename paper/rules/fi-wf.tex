\framebox{\(\gamma \turns \ty\)}

\inferrule [WF-Var]
{\alpha \in \gamma}
{\gamma \turns \alpha}

\inferrule [WF-Fun]
{\gamma \turns \ty_1 \andalso \gamma \turns \ty_2}
{\gamma \turns \ty_1 \to \ty_2}

\inferrule [WF-Forall]
{\gamma, \alpha \turns \ty}
{\gamma \turns \Forall \alpha \ty}

\inferrule [WF-And]
{\gamma \turns \ty_1 \andalso \gamma \turns \ty_2}
{\gamma \turns \ty_1 \Intersect \ty_2}

\inferrule [WF-Rec]
{\gamma \turns \ty}
{\gamma \turns \RecTy l \ty}
