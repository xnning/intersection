\framebox{\(\gamma \turns \tau\)}

\begin{mathpar}
\inferrule* [right=WF-Var]
{\alpha \in \gamma}
{\gamma \turns \alpha}

\inferrule* [right=WF-Fun]
{\gamma \turns \tau_1 \andalso \gamma \turns \tau_2}
{\gamma \turns \tau_1 \to \tau_2}

\inferrule* [right=WF-Forall]
{\gamma, \alpha \turns \tau}
{\gamma \turns \Forall \alpha \tau}

\inferrule* [right=WF-And]
{\gamma \turns \tau_1 \andalso \gamma \turns \tau_2}
{\gamma \turns \tau_1 \Intersect \tau_2}

\inferrule* [right=WF-Rec]
{\gamma \turns \tau}
{\gamma \turns \RecTy l \tau}
\end{mathpar}
