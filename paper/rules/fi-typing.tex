\framebox{\( \Gamma \turns e : t \yieldsnothing E \)}

\infrule[TrVar]
{(x,t) \in \Gamma}
{\Gamma \turns x : t \yieldsnothing x}

\infrule[TrAbs]
{\Gamma, \rel x t \turns e : t_1 \yieldsnothing E \andalso
 \Gamma \turns_{F} t}
{\Gamma \turns \abs {\rel x t} e : t \to t_1 \yieldsnothing {\abs {\rel x {\image t}} E}}

\infrule[TrTAbs]
{\Gamma, \alpha \turns e : t \yieldsnothing E}
{\Gamma \turns \Abs \alpha e : \forall \alpha. t \yieldsnothing {\Abs \alpha E}}

\infrule[TrApp]
{\Gamma \turns e_1 : t_1 \to t_2 \yieldsnothing {E_1} \\
 \Gamma \turns e_2 : t_3 \yieldsnothing {E_2} \\
 t_3 \subtype t_1 \yieldsnothing C}
{\Gamma \turns \app {e_1} {e_2} : t_2 \yieldsnothing {\app E_1 {(\app C E_2)}}}

\infrule[TrTApp]
{\Gamma \turns e : \forall \alpha. t_1 \yieldsnothing E \andalso
 \Gamma \turns_{F} t}
{\Gamma \turns \app e t : \subst \alpha t t_1 \yieldsnothing {\app E {\image t}}}

\infrule[TrMerge]
{\Gamma \turns e_1 : t_1 \yieldsnothing {E_1} \\
 \Gamma \turns e_2 : t_2 \yieldsnothing {E_2}}
{\Gamma \turns e_1 \dcomma e_2 : t_1 \with t_2 \yieldsnothing {\tupled {E_1, E_2}}}

\infrule[TrRcdIntro]
{\Gamma \turns e : t \yieldsnothing E}
{\Gamma \turns \recordintro l e : \recordtype l t \yieldsnothing E}

\infrule[TrRcdElim]
{\Gamma \turns e : t \yieldsnothing E \andalso
 \Gamma \turnsget (t; l) : t_1 \yieldsnothing C}
{\Gamma \turns e.l : t_1 \yieldsnothing {\app C E}}

\infrule[TrRcdUpd]
{\Gamma \turns e : t \yieldsnothing E \andalso
 \Gamma \turns e_1 : t_1 \yieldsnothing {E_1} \\
 \turnsput (t; l; e_1 : t_1 \yieldsnothing {E_1}) : (t_1', t') \yieldsnothing C \andalso
 t_1 \subtype t_1'}
{\Gamma \turns \recordupdate e l {e_1} : t' \yieldsnothing {\app C E}}
