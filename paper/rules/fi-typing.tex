\framebox{\( \Gamma \turns e : \ty \yieldsnothing E \)}

\inferrule [Var]
{(x,\ty) \in \Gamma}
{\Gamma \turns x : \ty}

\inferrule [Lam]
{\Gamma, x \hast \ty \turns e : \ty_1 \andalso \Gamma \turns \ty}
{\Gamma \turns \Lam x \ty e : \ty \to \ty_1}

\inferrule [BLam]
{\Gamma, \alpha \turns e : \ty}
{\Gamma \turns \BLam \alpha e : \Forall \alpha \ty}

\inferrule [App]
{\Gamma \turns e_1 : \ty_1 \to \ty_2 \\
 \Gamma \turns e_2 : \ty_3 \andalso \ty_3 \subtype \ty_1}
{\Gamma \turns \App {e_1} {e_2} : \ty_2}

\inferrule [TApp]
{\Gamma \turns e : \forall \alpha. \ty_1 \andalso \Gamma \turns \ty}
{\Gamma \turns \App e \ty : \subst \alpha \ty \ty_1}

\inferrule [Merge]
{\Gamma \turns e_1 : \ty_1 \andalso \Gamma \turns e_2 : \ty_2}
{\Gamma \turns e_1 \Merge e_2 : \ty_1 \Intersect \ty_2}

\inferrule [RecCon]
{\Gamma \turns e : \ty}
{\Gamma \turns \RecCon l e : \RecTy l \ty}

\inferrule [RecProj]
{\Gamma \turns e : \ty \andalso \Gamma \turnsget (\ty; l) : \ty_1}
{\Gamma \turns e.l : \ty_1}

\inferrule [RecUpd]
{\Gamma \turns e : \ty \andalso \Gamma \turns e_1 : \ty_1 \\
 \turnsput (t; l; e_1 : \ty_1) : (\ty_2, \ty_3) \andalso \ty_1 \subtype \ty_2}
{\Gamma \turns \RecUpd e l {e_1} : \ty_3}
