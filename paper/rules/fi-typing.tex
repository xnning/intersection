\framebox{\( \Gamma \vdash e : t \yieldsnothing E \)}

\infrule[TrVar]
{(x,t) \in \Gamma}
{\Gamma \vdash x : t \yieldsnothing x}

\infrule[TrAbs]
{\Gamma, \rel x t \vdash e : t_1 \yieldsnothing E \andalso
 \Gamma \vdash_{F} t}
{\Gamma \vdash \abs {\rel x t} e : t \to t_1 \yieldsnothing {\abs {\rel x {\imageof t}} E}}

\infrule[TrTAbs]
{\Gamma, \alpha \vdash e : t \yieldsnothing E}
{\Gamma \vdash \Abs \alpha e : \forall \alpha. t \yieldsnothing {\Abs \alpha E}}

\infrule[TrApp]
{\Gamma \vdash e_1 : t_1 \to t_2 \yieldsnothing {E_1} \\
 \Gamma \vdash e_2 : t_3 \yieldsnothing {E_2} \\
 t_3 \subtype t_1 \yieldsnothing C}
{\Gamma \vdash \app {e_1} {e_2} : t_2 \yieldsnothing {\app E_1 {(\app C E_2)}}}

\infrule[TrTApp]
{\Gamma \vdash e : \forall \alpha. t_1 \yieldsnothing E \andalso
 \Gamma \vdash_{F} t}
{\Gamma \vdash \app e t : \subst \alpha t t_1 \yieldsnothing {\app E {\imageof t}}}

\infrule[TrMerge]
{\Gamma \vdash e_1 : t_1 \yieldsnothing {E_1} \\
 \Gamma \vdash e_2 : t_2 \yieldsnothing {E_2}}
{\Gamma \vdash e_1 \dcomma e_2 : t_1 \with t_2 \yieldsnothing {\tupled {E_1, E_2}}}

\infrule[TrRcdIntro]
{\Gamma \vdash e : t \yieldsnothing E}
{\Gamma \vdash \recordintro l e : \recordtype l t \yieldsnothing E}

\infrule[TrRcdElim]
{\Gamma \vdash e : t \yieldsnothing E \andalso
 \Gamma \vdash_{get} (t; l) : t_1 \yieldsnothing C}
{\Gamma \vdash e.l : t_1 \yieldsnothing {\app C E}}

\infrule[TrRcdUpd]
{\Gamma \vdash e : t \yieldsnothing E \andalso
 \Gamma \vdash_{put} (t; l; e_1) : t' \yieldsnothing C}
{\Gamma \vdash \recordupdate e l {e_1} : t' \yieldsnothing {\app C E}}
