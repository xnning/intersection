\begin{mathpar}
\framebox{$\gamma \turns e : \tau$}

\inferrule* [right=$\rulelabelevar$]
{(x,\tau) \in \gamma}
{\judgee \gamma x \tau}

\inferrule* [right=$\rulelabelelam$]
{\judgee {\gamma, x \hast \tau} e {\tau_1} \andalso \gamma \turns \tau}
{\judgee \gamma {\lam x \tau e} {\tau \to \tau_1}}

\inferrule* [right=$\rulelabeleapp$]
{\judgee \gamma {e_1} {\tau_1 \to \tau_2} \\
 \judgee \gamma {e_2} {\tau_3} \andalso \tau_3 \subtype \tau_1}
{\judgee \gamma {\app {e_1} {e_2}} {\tau_2}}

\inferrule* [right=$\rulelabeleblam$]
{\judgee {\gamma, \alpha} e \tau}
{\judgee \gamma {\blam \alpha e} {\for \alpha \tau}}

\inferrule* [right=$\rulelabeletapp$]
{\judgee \gamma e {\for \alpha \tau_1} \andalso \gamma \turns \tau}
{\judgee \gamma {\tapp e \tau} {\subst \tau \alpha \tau_1}}

\inferrule* [right=$\rulelabelemerge$]
{\judgee \gamma {e_1} {\tau_1} \andalso \judgee \gamma {e_2} {\tau_2}}
{\judgee \gamma {e_1 \mergeop e_2} {\tau_1 \andop \tau_2}}

\inferrule* [right=$\rulelabelereccon$]
{\judgee \gamma e \tau}
{\judgee \gamma {\reccon l e} {\recty l \tau}}

\inferrule* [right=$\rulelabelerecproj$]
{\judgee \gamma e \tau \andalso \judgeget \tau l {\tau_1}}
{\judgee \gamma {e.l} {\tau_1}}

\inferrule* [right=$\rulelabelerecupd$]
{\judgee \gamma e \tau \andalso \judgee \gamma {e_1} {\tau_1} \\
 \judgeput \tau l {\tau_1} {\tau_2} {\tau_3} \andalso \tau_1 \subtype \tau_3}
{\judgee \gamma {\recupd e l {e_1}} {\tau_2}}
\end{mathpar}
