\begin{mathpar}
\framebox{$\gamma \turns e : \tau$}

\inferrule* [right=$\ruleevar$]
{(x,\tau) \in \gamma}
{\gamma \turns x : \tau}

\inferrule* [right=$\ruleelam$]
{\gamma, x \hast \tau \turns e : \tau_1 \andalso \gamma \turns \tau}
{\gamma \turns \Lam x \tau e : \tau \to \tau_1}

\inferrule* [right=$\ruleeapp$]
{\gamma \turns e_1 : \tau_1 \to \tau_2 \\
 \gamma \turns e_2 : \tau_3 \andalso \tau_3 \subtype \tau_1}
{\gamma \turns \App {e_1} {e_2} : \tau_2}

\inferrule* [right=$\ruleeblam$]
{\gamma, \alpha \turns e : \tau}
{\gamma \turns \BLam \alpha e : \Forall \alpha \tau}

\inferrule* [right=$\ruleetapp$]
{\gamma \turns e : \forall \alpha. \tau_1 \andalso \gamma \turns \tau}
{\gamma \turns \TApp e \tau : \subst \alpha \tau \tau_1}

\inferrule* [right=$\ruleemerge$]
{\gamma \turns e_1 : \tau_1 \andalso \gamma \turns e_2 : \tau_2}
{\gamma \turns e_1 \Merge e_2 : \tau_1 \Intersect \tau_2}

\inferrule* [right=$\ruleereccon$]
{\gamma \turns e : \tau}
{\gamma \turns \RecCon l e : \RecTy l \tau}

\inferrule* [right=$\ruleerecproj$]
{\gamma \turns e : \tau \andalso \judgeget \tau l {\tau_1}}
{\gamma \turns e.l : \tau_1}

\inferrule* [right=$\ruleerecupd$]
{\gamma \turns e : \tau \andalso \gamma \turns e_1 : \tau_1 \\
 \judgeput \tau l {\tau_1} {\tau_2} {\tau_3} \andalso \tau_1 \subtype \tau_3}
{\gamma \turns \RecUpd e l {e_1} : \tau_2}
\end{mathpar}
