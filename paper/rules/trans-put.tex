\infax[PutBase]
{\turnsput (\recordtype l \ty; l; e : \ty' \yields E) : (\ty, \recordtype l {\ty'})
  \yields {\absty x {\image {\recordtype l \ty}} E}}

  \yields {\absty x {\image {\ty_1 \with \ty_2}} {C (\fst ~ x)}}}

\infrule[PutRight]

  \yields {\absty x {\image {\ty_1 \with \ty_2}} {C (\snd ~ x)}}}
