\newcommand{\wfvar}{
  \inferrule* [right=WFVar]
    {\alpha * A \in \Gamma}
    {\istype \Gamma \alpha}
}

\newcommand{\wfbot}{
  \inferrule* [right=WFBot]
    { }
    {\istype \Gamma \bot}
}

\newcommand{\wffun}{
  \inferrule* [right=WFFun]
    {\istype \Gamma A \\ \istype \Gamma B}
    {\istype \Gamma {A \to B}}
}

\newcommand{\wfforall}{
  \inferrule* [right=WFForall]
    {\istype \Gamma A \\ \istype {\Gamma, \alpha \disjoint A} B}
    {\istype \Gamma {\for {\alpha \disjoint A} B}}
}

\newcommand{\wfinter}{
  \inferrule* [right=WFInter]
    {\istype \Gamma A \\ \istype \Gamma B \\ \isdisjoint \Gamma A B}
    {\istype \Gamma {A \inter B}}
}
