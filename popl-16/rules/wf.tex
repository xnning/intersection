\newcommand{\wfvar}{
  \inferrule* [right=WF\_Var]
    {\alpha * A \in \Gamma}
    {\istype \Gamma \alpha}
}

\newcommand{\wfbot}{
  \inferrule* [right=WF\_Bot]
    { }
    {\istype \Gamma \bot}
}

\newcommand{\wffun}{
  \inferrule* [right=WF\_Fun]
    {\istype \Gamma A \\ \istype \Gamma B}
    {\istype \Gamma {A \to B}}
}

\newcommand{\wfforall}{
  \inferrule* [right=WF\_Forall]
    {\istype \Gamma A \\ \istype {\Gamma, \alpha \disjoint A} B}
    {\istype \Gamma {\for {(\alpha \disjoint A)} B}}
}

\newcommand{\wfinterprim}{
  \inferrule* [right=WF\_Inter]
    {\istype \Gamma A \\ \istype \Gamma B}
    {\istype \Gamma {A \inter B}}
}

\newcommand{\wfinter}{
  \inferrule* [right=WF\_Inter]
    {\istype \Gamma A \\
     \istype \Gamma B \\
     \highlight{$\isdisjoint \Gamma A B$}}
    {\istype \Gamma {A \inter B}}
}
