\newcommand{\rulelabeltarget}{\text{T\_}}

\newcommand{\rulelabeltargetvar}{\rulelabeltarget\text{var}}
\newcommand{\ruletargetvar} {
\inferrule* [right=$\rulelabeltargetvar$]
  {(x,T) \in \Gamma}
  {\hastype \Gamma x T}
}

\newcommand{\rulelabeltargetlam}{\rulelabeltarget\text{Lam}}
\newcommand{\ruletargetlam} {
\inferrule* [right=$\rulelabeltargetlam$]
  {\hastype {\Gamma, x \hast T} E {T_1} \andalso \istype \Gamma T}
  {\hastype \Gamma {\lamty x T E} {T \to T_1}}
}

\newcommand{\rulelabeltargetApp}{\rulelabeltarget\text{App}}
\newcommand{\ruletargetapp}{
\inferrule* [right=$\rulelabeltargetApp$]
  {\hastype \Gamma {E_1} {T_1 \to T_2} \andalso \hastype \Gamma {E_2} {T_1}}
  {\hastype \Gamma {\app {E_1} {E_2}} {T_2}}
}

\newcommand{\rulelabeltargetBLam}{\rulelabeltarget\text{BLam}}
\newcommand{\ruletargetblam}{
\inferrule* [right=$\rulelabeltargetBLam$]
  {\hastype {\Gamma, \alpha} E T}
  {\hastype \Gamma {\blam \alpha E} {\for \alpha T}}
}

\newcommand{\rulelabeltargetTApp}{\rulelabeltarget\text{TApp}}
\newcommand{\ruletargettapp}{
\inferrule* [right=$\rulelabeltargetTApp$]
  {\hastype \Gamma E {\for \alpha {T_1}} \andalso \istype \Gamma T}
  {\hastype \Gamma {\tapp E T} {\subst T \alpha T_1}}
}

\newcommand{\rulelabeltargetPair}{\rulelabeltarget\text{Pair}}
\newcommand{\ruletargetpair}{
\inferrule* [right=$\rulelabeltargetPair$]
  {\hastype \Gamma {E_1} {T_1} \andalso \hastype \Gamma {E_2} {T_2}}
  {\hastype \Gamma {\pair {E_1} {E_2}} {\pair {T_1} {T_2}}}
}

\newcommand{\rulelabeltargetprojl}{\rulelabeltarget\text{Proj}_1}
\newcommand{\ruletargetprojl}{
\inferrule* [right=$\rulelabeltargetprojl$]
  {\hastype \Gamma E {\pair {T_1} {T_2}}}
  {\hastype \Gamma {\proj 1 E} {T_1}}
}

\newcommand{\rulelabeltargetprojr}{\rulelabeltarget\text{Proj}_2}
\newcommand{\ruletargetprojr}{
\inferrule* [right=$\rulelabeltargetprojr$]
  {\hastype \Gamma E {\pair {T_1} {T_2}}}
  {\hastype \Gamma {\proj 2 E} {T_2}}
}
