\newcommand{\ruleLabelSub}{\text{Sub}}
\newcommand{\ruleLabelsubvar}{\ruleLabelSub\text{Var}}
\newcommand{\ruleLabelSubTop}{\ruleLabelSub\text{Top}}
\newcommand{\ruleLabelsubfun}{\ruleLabelSub\text{Fun}}
\newcommand{\ruleLabelsubforall}{\ruleLabelSub\text{Forall}}
\newcommand{\ruleLabelsubinter}{\ruleLabelSub\text{And}}
\newcommand{\ruleLabelsubinterLeft}{\ruleLabelSub{\text{And}_1}}
\newcommand{\ruleLabelsubinterRight}{\ruleLabelSub{\text{And}_2}}

\newcommand{\subvar}{
\inferrule* [right=$\ruleLabelsubvar$]
  { }
  {\alpha \subtype \alpha \yields {\lam x {\im \alpha} x}}
}

\newcommand{\subTop}{
\inferrule* [right=$\ruleLabelSubTop$]
  { }
  {A \subtype \top \yields {\lam x {\im A} {()}}}
}

\newcommand{\subfun}{
\inferrule* [right=$\ruleLabelsubfun$]
  {A_3 \subtype A_1 \yields {C_1} \\ A_2 \subtype A_4 \yields {C_2}}
  {A_1 \to A_2 \subtype A_3 \to A_4
  \yields
      {\lam f {\im {A_1 \to A_2}}
      {\lam x {\im {A_3}}
          {\app {C_2} {(\app f {(\app {C_1} x)})}}}}}
}

\newcommand{\subforall}{
\inferrule* [right=$\ruleLabelsubforall$]
  {A_1 \subtype \subst {\alpha_1} {\alpha_2} A_2 \yields C}
  {\for {\alpha_1 * A_3} A_1 \subtype \for {\alpha_2 * A_3} A_2
    \yields
      {\lam f {\im {\for {\alpha_1 * A_3} A_1}}
        {\blam \alpha {\app C {(\app f \alpha)}}}}}
}

\newcommand{\subinter}{
\inferrule* [right=$\ruleLabelsubinter$]
  {A_1 \subtype A_2 \yields {C_1} \\ A_1 \subtype A_3 \yields {C_2}}
  {A_1 \subtype A_2 \inter A_3
    \yields
      {\lam x {\im {A_1}}
        {\pair {\app {C_1} x} {\app {C_2} x}}}}
}

\newcommand{\subinterleft}{
\inferrule* [right=$\ruleLabelsubinterLeft$]
  {A_1 \subtype A_3 \yields C}
  {A_1 \inter A_2 \subtype A_3
    \yields
      {\lam x {\im {A_1 \inter A_2}}
        {\app C {(\proj 1 x)}}}}
}

\newcommand{\subinterright}{
\inferrule* [right=$\ruleLabelsubinterRight$]
  {A_2 \subtype A_3 \yields C}
  {A_1 \inter A_2 \subtype A_3
    \yields
      {\lam x {\im {A_1 \inter A_2}}
        {\app C {(\proj 2 x)}}}}
}

\newcommand{\subRec}{
\inferrule* [right=$\ruleLabelSubRec$]
  {A_1 \subtype A_2 \yields C}
  {\recordType l {A_1} \subtype \recordType l {A_2}
    \yields
      {\lam x {\im {\recordType l {A_1}}} {\app C x}}}
}
